    This is an integration test that performs an ACTUAL render. It must:
    1. Bundle the composition (or reuse the bundle if available)
    2. Call `selectComposition()` + `renderMedia()` with Hebrew + English test content
    3. Verify the output MP4 exists and has correct properties

    Test structure:
    ```typescript
    import { describe, it, expect, beforeAll } from "vitest";
    import { bundle } from "@remotion/bundler";
    import { selectComposition, renderMedia } from "@remotion/renderer";
    import path from "path";
    import fs from "fs";
    import { execSync } from "child_process";

    describe("Smoke Render", () => {
      let bundlePath: string;
      const outputPath = "/tmp/renders/smoke-test.mp4";
      const testVideoPath = path.join(__dirname, "../../test-assets/sample.mp4");

      beforeAll(async () => {
        // Bundle once for all tests (this takes 15-30s)
        bundlePath = await bundle({
          entryPoint: path.join(__dirname, "../../remotion/index.ts"),
          webpackOverride: (config) => config,
        });
      }, 60000); // 60s timeout for bundling

      it("REND-01: renders a valid MP4 with correct format", async () => {
        const composition = await selectComposition({
          serveUrl: bundlePath,
          id: "ReelTemplate",
          inputProps: {
            sourceVideoUrl: "", // not used directly by composition
            sourceVideoLocalPath: testVideoPath,
            hookText: "שלום עולם",
            bodyText: "זה טקסט לבדיקה עם English words בתוך משפט",
            textDirection: "rtl",
            animationStyle: "fade",
            durationInSeconds: 5,
          },
        });

        // Override duration to 5 seconds (150 frames) for fast smoke test
        await renderMedia({
          composition: { ...composition, durationInFrames: 150 },
          serveUrl: bundlePath,
          codec: "h264",
          outputLocation: outputPath,
          inputProps: {
            sourceVideoUrl: "",
            sourceVideoLocalPath: testVideoPath,
            hookText: "שלום עולם",
            bodyText: "זה טקסט לבדיקה עם English words בתוך משפט",
            textDirection: "rtl",
            animationStyle: "fade",
            durationInSeconds: 5,
          },
          concurrency: 1,
        });

        // Verify file exists
        expect(fs.existsSync(outputPath)).toBe(true);

        // Verify file is non-trivial size (> 10KB)
        const stats = fs.statSync(outputPath);
        expect(stats.size).toBeGreaterThan(10000);

        // Use ffprobe to verify format (if available)
        try {
          const probe = execSync(
            `ffprobe -v quiet -print_format json -show_streams "${outputPath}"`,
            { encoding: "utf-8" }
          );
          const info = JSON.parse(probe);
          const videoStream = info.streams.find(s => s.codec_type === "video");
          const audioStream = info.streams.find(s => s.codec_type === "audio");

          // REND-01 checks
          expect(videoStream.codec_name).toBe("h264");
          expect(Number(videoStream.width)).toBe(1080);
          expect(Number(videoStream.height)).toBe(1920);
          // fps check: r_frame_rate is "30/1" or avg_frame_rate
          const fps = eval(videoStream.r_frame_rate); // "30/1" -> 30
          expect(fps).toBe(30);

          if (audioStream) {
            expect(audioStream.codec_name).toBe("aac");
          }
        } catch (e) {
          // ffprobe not available — file existence and size checks are sufficient
          console.warn("ffprobe not available — skipping detailed format checks");
        }
      }, 120000); // 2 minute timeout for render

      afterAll(() => {
        // Cleanup rendered file
        try { fs.unlinkSync(outputPath); } catch {}
      });
    });
    ```

    - Test completes in under 2 minutes
